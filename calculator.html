            }


            // Traditional Realtor Costs

            let traditionalCommission = currentHomePrice * 0.06;

            let eliteCommissionRefund = currentHomePrice * 0.03;

            let homeInspectionValue = 700; // Estimated average


            let eliteTotalSavings = eliteCommissionRefund + eliteNewHomeRebate + eliteClosingCredit + homeInspectionValue;

            let eliteNet = currentHomePrice - (traditionalCommission - eliteTotalSavings);


            document.getElementById('results').innerHTML = `
